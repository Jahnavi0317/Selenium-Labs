        // Step 2: Verify Title of the page
        String pageTitle = driver.getTitle();
        System.out.println("Page Title: " + pageTitle);

        // Step 3: Click on 'My Account' dropdown
        WebElement myAccount = driver.findElement(By.xpath("//span[text()='My Account']"));
        myAccount.click();

        // Step 4: Select 'Register' from dropdown
        WebElement register = driver.findElement(By.linkText("Register"));
        register.click();

        // Step 5: Verify 'Register Account' page heading

        // Step 6: Click on 'Continue' button without entering details
        WebElement continueBtn = driver.findElement(By.xpath("//input[@value='Continue']"));
        continueBtn.click();

        // Step 7: Verify warning message
        WebElement warning = wait.until(ExpectedConditions.visibilityOfElementLocated(By.xpath("//div[contains(@class,'alert-danger')]")));
        System.out.println("Warning Message: " + warning.getText());

        // Close the browser
        driver.quit();
    }
}
